%I/P: Function, lower limit, upper limit, n
function[]=PSJ_SIMP3(fun,x0,xn,n)
h=(xn-x0)/n;
y0=feval(fun,x0);
yn=feval(fun,xn);
yr=0;
ys=0;
for i=1:1:n-1
yr=yr+feval(fun,x0+i*h);
end
for j=3:3:n-1
ys=ys+feval(fun,x0+j*h);
end
yt=yr-ys;
I=(3*h/8)*(y0+yn+2*ys+3*yt);
h
I
